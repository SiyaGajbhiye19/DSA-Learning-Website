Pop in Stack

Code

        
        #include <stdio.h> 
        #include <stdlib.h>
            #define MAX 100   
                struct Stack {
                    int arr[MAX];
                    int top;
                };
                
                void initStack(struct Stack* s) {
                    s->top = -1;
                }
                
                int isFull(struct Stack* s) {
                    return s->top == MAX - 1;
                }
                
                int isEmpty(struct Stack* s) {
                    return s->top == -1;
                }
                
                void push(struct Stack* s, int value) {
                    if (isFull(s)) {
                        printf("Stack overflow! Cannot push %d\n", value);
                        return;
                    }
                    s->arr[++(s->top)] = value;
                }
                
                int pop(struct Stack* s) {
                    if (isEmpty(s)) {
                        printf("Stack underflow! The stack is empty.\n");
                        return -1;
                    }
                    return s->arr[(s->top)--];
                }
                
                int peek(struct Stack* s) {
                    if (isEmpty(s)) {
                        printf("The stack is empty.\n");
                        return -1;
                    }
                    return s->arr[s->top];
                }
                
                void printStack(struct Stack* s) {
                    if (isEmpty(s)) {
                        printf("The stack is empty.\n");
                        return;
                    }
                    printf("Stack contents: ");
                    for (int i = 0; i <= s->top; i++) {
                        printf("%d ", s->arr[i]);
                    }
                    printf("\n");
                }
                
                int main() {
                    struct Stack stack;
                    initStack(&stack);  
                
                    int n, value;
                
                    printf("Enter the number of elements to push onto the stack: ");
                    scanf("%d", &n);
                
                    for (int i = 0; i < n; i++) {
                        printf("Enter element %d: ", i + 1);
                        scanf("%d", &value);
                        push(&stack, value);
                    }
                
                    printStack(&stack);
                
                    int popCount;
                    printf("Enter how many elements to pop: ");
                    scanf("%d", &popCount);
                
                    for (int i = 0; i < popCount; i++) {
                        pop(&stack);
                    }
                
                    printStack(&stack);
                
                    return 0;
                }

Description

The stack::pop() function to remove or ‘pop’ the element from the top of stack. As we are only inserting and removing from the top, the most recently inserted element will be removed first.

Steps:

  • Syntax: st.pop();
  • Parameters: This function does not take any parameter.
  • Return Value: This function does not return any value.
  • Time Complexity: O(1)

Example

Input:

Enter the number of elements to push onto the stack: 4
            Enter element 1: 1
            Enter element 2: 2
            Enter element 3: 3
            Enter element 4: 4
            Stack contents: 1 2 3 4
            Enter how many elements to pop: 2

Step-by-Step Process of Pop Stack:

Steps:

  • The program first asks how many elements to push onto the stack (4 elements).
  • The program then accepts the elements 1, 2, 3, 4 and pushes them onto the stack.
  • It prints the contents of the stack (1 2 3 4).
  • It then asks how many elements to pop (2 elements).
  • It pops 2 elements from the stack, and the stack contents are printed again (1 2).

Output:

Stack contents: 1 2 3 4
            Stack contents: 1 2
