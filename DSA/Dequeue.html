DEQUEUE

Code

        
        #include <stdio.h> 
        #include <stdlib.h>
            struct Queue {
                int front, rear, size;
                unsigned capacity;
                int* array;
            };
            
            struct Queue* createQueue(unsigned capacity) {
                struct Queue* queue = (struct Queue*)malloc(sizeof(struct Queue));
                queue->capacity = capacity;
                queue->front = 0;  
                queue->size = 0;
                queue->rear = capacity - 1;
                queue->array = (int*)malloc(queue->capacity * sizeof(int));
                return queue;
            }
            
            int isFull(struct Queue* queue) {
                return (queue->size == queue->capacity);
            }
            
            int isEmpty(struct Queue* queue) {
                return (queue->size == 0);
            }
            
            void enqueue(struct Queue* queue, int item) {
                if (isFull(queue)) {
                    printf("Queue is full! Cannot enqueue %d\n", item);
                    return;
                }
                queue->rear = (queue->rear + 1) % queue->capacity;
                queue->array[queue->rear] = item;
                queue->size++;
                printf("%d enqueued to queue\n", item);
            }
            
            int dequeue(struct Queue* queue) {
                if (isEmpty(queue)) {
                    printf("\nQueue is empty, cannot dequeue\n");
                    return -1;
                }
                int item = queue->array[queue->front];
                queue->front = (queue->front + 1) % queue->capacity;
                queue->size--;
                return item;
            }
            
            void printQueue(struct Queue* queue) {
                if (isEmpty(queue)) {
                    printf("Queue is empty!\n");
                    return;
                }
                printf("Queue elements: ");
                for (int i = 0; i < queue->size; i++) {
                    printf("%d ", queue->array[(queue->front + i) % queue->capacity]);
                }
                printf("\n");
            }
            
            int main() {
                int capacity, num, value, dequeueCount;
            
                printf("Enter the capacity of the queue: ");
                scanf("%d", &capacity);
            
                struct Queue* queue = createQueue(capacity);
            
                printf("Enter the number of elements to enqueue: ");
                scanf("%d", &num);
            
                if (num > capacity) {
                    printf("You cannot enqueue more elements than the queue capacity!\n");
                    return 0;
                }
            
                for (int i = 0; i < num; i++) {
                    printf("Enter element %d: ", i + 1);
                    scanf("%d", &value);
                    enqueue(queue, value);
                }
            
                printQueue(queue);
            
                printf("Enter the number of elements to dequeue: ");
                scanf("%d", &dequeueCount);
            
                for (int i = 0; i < dequeueCount; i++) {
                    int removed = dequeue(queue);
                    if (removed != -1) {
                        printf("Dequeued: %d\n", removed);
                    }
                }
            
                printQueue(queue);
            
                free(queue->array);
                free(queue);
            
                return 0;
            }

Description

In Data Structures and Algorithms (DSA), dequeue refers to the operation of removing (deleting) an element from the front of a queue.

Steps:

  • Check if the queue is empty.
  • If the queue is empty, return the underflow error and exit.
  • If the queue is not empty, access the data where the front is pointing.
  • Increment the front pointer to point to the next available data element.
  • return success.
    • Complexity Analysis:
            Time Complexity: O(1)
            Space Complexity: O(N)

Example

Input:

Enter the capacity of the queue: 5
            Enter the number of elements to enqueue: 4
            Enter element 1: 10
            Enter element 2: 20
            Enter element 3: 30
            Enter element 4: 40

Step-by-Step Process of Dequeue:

User Input:

  • User enters queue capacity (e.g., 5).
  • User enters elements to enqueue (e.g., 10, 20, 30, 40).
  • Queue elements are displayed.
  • User enters number of elements to dequeue (e.g., 2).
  • Dequeue operation occurs, and removed elements are displayed.
  • Updated queue is displayed.

Dequeue Operations:

  • 10 added → Queue: [10]
  • 20 added → Queue: [10, 20]
  • 30 added → Queue: [10, 20, 30]
  • 40 added → Queue: [10, 20, 30, 40]

Output after Enqueue:

Queue elements: 10 20 30 40

Output after Dequeue:

Dequeued: 10
            Dequeued: 20
            Queue elements: 30 40
