Deletion at the begin in Circular Linked List

Code

        
        #include <stdio.h> 
            #include <stdlib.h>
                struct Node {
                    int data;
                    struct Node* next;
                };
                
                struct Node* createNode(int value) {
                    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
                    newNode->data = value;
                    newNode->next = NULL;
                    return newNode;
                }
                
                struct Node* insertEnd(struct Node* last, int value) {
                    struct Node* newNode = createNode(value);
                    if (last == NULL) {
                        newNode->next = newNode;
                        return newNode;
                    }
                    
                    newNode->next = last->next;
                    last->next = newNode;
                    return newNode;
                }
                
                struct Node* deleteFirstNode(struct Node* last) {
                    if (last == NULL) {
                        printf("List is empty\n");
                        return NULL;
                    }
                
                    struct Node* head = last->next;
                
                    if (head == last) {
                        free(head);
                        return NULL;
                    } else {
                        last->next = head->next;
                        free(head);
                    }
                
                    return last;
                }
                
                void printList(struct Node* last) {
                    if (last == NULL) {
                        printf("List is empty.\n");
                        return;
                    }
                
                    struct Node* head = last->next;
                    do {
                        printf("%d ", head->data);
                        head = head->next;
                    } while (head != last->next);
                    printf("\n");
                }
                
                int main() {
                    struct Node* last = NULL;
                    int n, data;
                
                    printf("Enter the number of nodes: ");
                    scanf("%d", &n);
                
                    if (n <= 0) {
                        printf("Invalid number of nodes!\n");
                        return 0;
                    }
                
                    for (int i = 0; i < n; i++) {
                        printf("Enter value for node %d: ", i + 1);
                        scanf("%d", &data);
                        last = insertEnd(last, data);
                    }
                
                    printf("Original Circular Linked List: ");
                    printList(last);
                
                    last = deleteFirstNode(last);
                
                    printf("After deleting first node: ");
                    printList(last);
                
                    return 0;
                }

Description

To delete the first node of a circular linked list, we first check if the list is empty. I

Steps:

  • Check if the list is empty → If head == NULL, return NULL (nothing to delete).
  • Check if there is only one node → If head->next == head, free the node and return NULL.
  • Update head and last node: Set head = head->next (move head to the next node). Find the last node (tail), update tail->next = head.
  • Free memory of the deleted node
  • Return the updated head pointer.

Example

Input:

User enters 4 nodes: 10 → 20 → 30 → 40 (Circular)

Step-by-Step Process of deletion at the begin in Circular Linked List:

Step 1: Create Circular Linked List

  • 10 → 20 → 30 → 40 → (back to 10)

Step 2: Delete First Node (10)

  • Update last->next to point to head->next (20).
  • Free memory of node 10.

Step 3: Updated List

  • 20 → 30 → 40 → (back to 20)

Output:

Original Circular Linked List: 10 20 30 40  
            After deleting first node: 20 30 40
